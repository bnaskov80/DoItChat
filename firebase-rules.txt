rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    match /users/{userId} {
      allow read: if signedIn();
      allow create: if signedIn() && (
        request.auth.uid == userId ||
        (userId == 'user2' && request.resource.data.id == 'user2')
      );
      allow update: if signedIn() && (
        request.auth.uid == userId ||
        userId == 'user2'
      );
      allow delete: if false;
    }

    match /channels/{channelId} {
      allow read: if signedIn();
      allow create: if signedIn();
      allow update: if signedIn();
      allow delete: if signedIn() && resource.data.createdBy == request.auth.uid;
    }

    match /messages/{messageId} {
      allow read: if signedIn();
      allow create: if signedIn();
      allow update: if signedIn();
      allow delete: if signedIn() && (
        resource.data.userId == request.auth.uid ||
        (
          exists(/databases/$(database)/documents/channels/$(resource.data.channelId)) &&
          get(/databases/$(database)/documents/channels/$(resource.data.channelId)).data.createdBy == request.auth.uid
        )
      );
    }
  }
}
